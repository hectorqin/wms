## 项目架构

本系统采用 **MVC + Service** 架构模式：

```
┌─────────────────────────────────────────────┐
│                  Controller                 │
│              (控制器层 - C)                  │
└─────────────────┬───────────────────────────┘
                  │
┌─────────────────▼───────────────────────────┐
│                   Service                   │
│              (业务逻辑层 - S)                │
└─────────────────┬───────────────────────────┘
                  │
┌─────────────────▼───────────────────────────┐
│                    Model                    │
│               (数据模型层 - M)               │
└─────────────────┬───────────────────────────┘
                  │
┌─────────────────▼───────────────────────────┐
│                Database                      │
│                 (数据库)                     │
└─────────────────────────────────────────────┘
```

## 开发规范

### 命名规范

| 类型 | 规范 | 示例 |
|------|------|------|
| 控制器 | 驼峰命名，首字母大写 | `User.php`, `ProductType.php` |
| 模型 | 驼峰命名，首字母大写 | `User.php`, `ProductType.php` |
| 服务层 | 驼峰命名，以Service结尾 | `UserService.php` |
| 验证器 | 驼峰命名，以Validate结尾 | `UserValidate.php` |
| 视图 | 小写加下划线 | `user_index.html`, `create.html` |
| 方法 | 驼峰命名 | `getUserList()`, `saveData()` |

### 代码示例

```php
// 控制器命名
namespace app\controller;
use app\service\UserService;

class User extends Base
{
    protected $service;
    
    public function __construct(UserService $service)
    {
        parent::__construct();
        $this->service = $service;
    }
    
    public function index()
    {
        $this->assign($this->service->getVar());
        $this->assign(['list' => $this->service->page()]);
        return view();
    }
}
```

## 模块开发

### 新增模块步骤

以添加 `Customer`（客户管理）模块为例：

#### 1. 创建数据库表

```sql
CREATE TABLE `w_customer` (
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `name` varchar(100) NOT NULL,
  `phone` varchar(20) DEFAULT NULL,
  `address` varchar(255) DEFAULT NULL,
  `status` tinyint(1) DEFAULT 1,
  `add_time` int(11) DEFAULT NULL,
  PRIMARY KEY (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
```

#### 2. 创建模型

`app/model/Customer.php`

```php
<?php
namespace app\model;
use think\Model;

class Customer extends Model
{
    protected $pk = 'id';
    protected $autoWriteTimestamp = 'int';
    
    // 关联关系
    public function findCompany()
    {
        return $this->hasOne('\app\model\Company', 'id', 'company_id');
    }
}
```

#### 3. 创建验证器

`app/validate/CustomerValidate.php`

```php
<?php
namespace app\validate;
use think\Validate;

class CustomerValidate extends Validate
{
    protected $rule = [
        'name'  => 'require|max:100',
        'phone' => 'mobile',
    ];
    
    protected $message = [
        'name.require' => '客户名称不能为空',
        'phone.mobile' => '手机号格式不正确',
    ];
}
```

#### 4. 创建服务层

`app/service/CustomerService.php`

```php
<?php
namespace app\service;
use app\model\Customer;
use app\validate\CustomerValidate;

class CustomerService
{
    // 获取视图变量
    public function getVar()
    {
        return [
            'status_list' => [1 => '启用', 0 => '禁用'],
        ];
    }
    
    // 分页列表
    public function page()
    {
        $data = request()->get();
        $where = [];
        
        empty($data['name']) || $where['name'] = ['like', '%'.$data['name'].'%'];
        empty($data['status']) || $where['status'] = $data['status'];
        
        return Customer::where($where)->paginate(10);
    }
    
    // 保存数据
    public function save()
    {
        request()->isPost() || die('request not post!');
        
        $param = request()->param();
        $validate = new CustomerValidate();
        
        if (!$validate->check($param)) {
            return ['error' => 100, 'msg' => $validate->getError()];
        }
        
        $customer = new Customer();
        $customer->name = $param['name'];
        $customer->phone = $param['phone'] ?? '';
        $customer->address = $param['address'] ?? '';
        $customer->status = $param['status'] ?? 1;
        $customer->add_time = time();
        
        if ($customer->save()) {
            return ['error' => 0, 'msg' => '保存成功'];
        }
        return ['error' => 100, 'msg' => '保存失败'];
    }
    
    // 编辑数据
    public function edit($id)
    {
        return Customer::get($id);
    }
    
    // 更新数据
    public function update()
    {
        $param = request()->post();
        $validate = new CustomerValidate();
        
        if (!$validate->check($param)) {
            return ['error' => 100, 'msg' => $validate->getError()];
        }
        
        $customer = Customer::get($param['id']);
        $customer->name = $param['name'];
        $customer->phone = $param['phone'] ?? '';
        $customer->address = $param['address'] ?? '';
        $customer->status = $param['status'] ?? 1;
        
        if ($customer->save()) {
            return ['error' => 0, 'msg' => '更新成功'];
        }
        return ['error' => 100, 'msg' => '更新失败'];
    }
    
    // 删除数据
    public function delete($id)
    {
        if (Customer::destroy($id)) {
            return ['error' => 0, 'msg' => '删除成功'];
        }
        return ['error' => 100, 'msg' => '删除失败'];
    }
}
```

#### 5. 创建控制器

`app/controller/Customer.php`

```php
<?php
namespace app\controller;
use app\service\CustomerService;

class Customer extends Base
{
    protected $service;
    
    public function __construct(CustomerService $service)
    {
        parent::__construct();
        $this->service = $service;
    }
    
    public function index()
    {
        $this->assign($this->service->getVar());
        $this->assign(['list' => $this->service->page()]);
        return view();
    }
    
    public function create()
    {
        $this->assign($this->service->getVar());
        return view();
    }
    
    public function save()
    {
        return $this->service->save();
    }
    
    public function edit($id)
    {
        $this->assign($this->service->getVar());
        $this->assign(['info' => $this->service->edit($id)]);
        return view();
    }
    
    public function update()
    {
        return $this->service->update();
    }
    
    public function delete($id)
    {
        return $this->service->delete($id);
    }
}
```

## 数据库操作

### 常用查询

```php
// 获取单条
$user = User::get($id);
$user = User::where('id', $id)->find();

// 获取多条
$list = User::all();
$list = User::where('status', 1)->select();

// 分页
$list = User::paginate(10);

// 条件查询
$list = User::where('status', 1)
    ->where('name', 'like', '%test%')
    ->order('id desc')
    ->limit(10)
    ->select();

// 统计
$count = User::where('status', 1)->count();
$sum = Order::where('status', 1)->sum('amount');
```

### 常用写入

```php
// 新增
$user = new User();
$user->name = '张三';
$user->save();

// 批量新增
$data = [
    ['name' => '张三', 'age' => 20],
    ['name' => '李四', 'age' => 25],
];
User::insertAll($data);

// 更新
$user = User::get($id);
$user->name = '王五';
$user->save();

// 条件更新
User::where('status', 0)->update(['status' => 1]);

// 删除
User::destroy($id);
User::where('status', 0)->delete();
```

## 常用功能

### 文件上传

```php
public function upload()
{
    $file = request()->file('file');
    if ($file) {
        $info = $file->move(ROOT_PATH . 'public/uploads');
        if ($info) {
            return json(['error' => 0, 'msg' => '上传成功', 'data' => '/uploads/' . $info->getSaveName()]);
        }
        return json(['error' => 100, 'msg' => $file->getError()]);
    }
}
```

## 常见问题

### URL访问404

1. 检查 `public/.htaccess` 是否存在
2. 确认 `url_route_on` 配置
3. 清除 `runtime/cache` 目录

### 页面空白

1. 开启调试模式：`'app_debug' => true`
2. 检查 `runtime` 目录权限
3. 查看 `runtime/log` 目录日志

### 权限不生效

1. 检查 `app/config.php` 中的 `super_admin` 配置
2. 确认用户角色ID正确
3. 清除缓存
